import { Component, EventEmitter, Input, Output } from '@angular/core';
import { InscriptionGetAllForDriverItemI } from '@interfaces';

@Component({
    selector: 'inscriptions-state',
    templateUrl: 'inscriptions-state.component.html',
    styleUrls: ['./inscriptions-state.component.scss'],
})
export class InscriptionsStateComponent {
    @Input() title: string;
    @Input() items: InscriptionGetAllForDriverItemI[] = [];
    @Input() type: string;
    @Input() state = false;
    @Output() showOrHide: EventEmitter<string> = new EventEmitter<string>();
    @Output() openPopover: EventEmitter<{
        event: any;
        carId: string;
        tournamentId: string;
    }> = new EventEmitter();
}

<ng-container *ngIf="items && items.length > 0">
    <ion-card mode="ios">
        <ion-item
            (click)="state = !state"
            button
            (click)="showOrHide.emit(type)"
            detail="false"
            [lines]="!state ? 'none' : 'full'"
        >
            <ion-label
                >{{ title }} <small>({{ items.length }})</small></ion-label
            >
            <ion-icon
                slot="end"
                [name]="state ? 'chevron-up' : 'chevron-down'"
            ></ion-icon>
        </ion-item>
    </ion-card>

    <ng-container *ngIf="state">
        <inscriptions-item
            *ngFor="let item of items"
            [item]="item"
            (openPopover)="openPopover.emit($event)"
        >
        </inscriptions-item>
    </ng-container>
</ng-container>
